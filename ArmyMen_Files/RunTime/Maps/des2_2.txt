//INIT
IF startup THEN
	VULNERABLESARGE,
	ALLY GREEN BLUE,
	ALLY GREEN GRAY,
	TRIGGER ALLYG,
	TRIGGER TANAI,
	TRIGGER BYEBYEBLUE

//EndGames
IF killed_GreenSarge THEN
	TRIGGERDELAY 1 tanwins


IF INORDER get_GreenSarge_Key1 THEN
	INVULNERABLESARGE,
	TRIGGERDELAY 3 greenwins


//drop the key
IF killed_TanKey1 THEN
	REMOVEOBJECT Key1,
	PLACEOBJECT Key1 TanKey1

IF get_GreenSarge_Key1 THEN
	TRIGGERDELAY 3 GetBack

IF GetBack THEN
	PLAYSTREAM DESERT2_2M

IF ALLYG THEN
	SETAICONTROL GreenRifle1,
	SETAICONTROL GreenRifle2,
	SETAICONTROL GreenRifle101

IF killed_TanTank101 THEN
	REMOVEOBJECT Para1,
	PLACEOBJECT Para1 TanTank101

IF killed TanRifle107 THEN
	SETUSERCONTROL GreenRifle101

IF TANAI THEN
	DEPLOY TanTank101 ATMODE DEFEND,
	SETAICONTROL TanTank101,
	DEPLOY TanKey1 ATMODE DEFEND,
	SETAICONTROL TanKey1,
	DEPLOY TanTank102 ATMODE DEFEND,
	SETAICONTROL TanTank102

IF BYEBYEBLUE THEN
	ORDER BlueRifle1 GOTO TanTank101 INMODE ATTACK THEN ATMODE ATTACK,
	ORDER BlueRifle2 GOTO TanTank101 INMODE ATTACK THEN ATMODE ATTACK,
	ORDER BlueRifle3 GOTO TanTank101 INMODE ATTACK THEN ATMODE ATTACK

//ITEM DROPS
IF killed_TanBazooka1 THEN
	REMOVEOBJECT Boom1,
	PLACEOBJECT Boom1 TanBazooka1

IF killed_TanRifle48 THEN
	REMOVEOBJECT Health1,
	PLACEOBJECT Health1 TanRifle48

IF killed_TanRifle34 THEN
	REMOVEOBJECT Health2,
	PLACEOBJECT Health2 TanRifle34

IF killed_TanRifle23 THEN
	REMOVEOBJECT Health3,
	PLACEOBJECT Health3 TanRifle23
